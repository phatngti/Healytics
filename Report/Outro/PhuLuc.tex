\section*{1. Danh sách API MoMo đã tích hợp}
\begin{table}[H]
    \centering
    \caption{Các API MoMo đã tích hợp}
    \renewcommand{\arraystretch}{1.2}
    \begin{tabular}{|>{\raggedright\arraybackslash}p{3.2cm}|>{\raggedright\arraybackslash}p{4.2cm}|>{\raggedright\arraybackslash}p{1.7cm}|>{\raggedright\arraybackslash}p{4.2cm}|}
        \hline
        \textbf{Chức năng} & \textbf{Endpoint} & \textbf{Phương thức} & \textbf{Mô tả} \\
        \hline
        Tạo yêu cầu thanh toán & {\small\ttfamily /momo/payment/request} & POST & Tạo giao dịch thanh toán mới, trả về URL/mã QR để người dùng thanh toán qua MoMo app. \\
        \hline
        Xác thực giao dịch & {\small\ttfamily /momo/payment/confirm} & POST & Nhận callback từ MoMo xác nhận giao dịch thành công/thất bại, cập nhật trạng thái booking. \\
        \hline
        Truy vấn trạng thái giao dịch & {\small\ttfamily /momo/payment/status} & GET & Kiểm tra trạng thái thanh toán của một giao dịch dựa trên transactionId/orderId. \\
        \hline
    \end{tabular}
    \renewcommand{\arraystretch}{1}
\end{table}

\section*{2. Giải thích schema dữ liệu MoMo}

\subsection*{2.1. Format request/response JSON của các API MoMo tích hợp}

% --- Tạo yêu cầu thanh toán ---
\textbf{a) API: Tạo yêu cầu thanh toán}

\textit{Request gửi lên MoMo:}
\begin{table}[H]
    \centering
    \caption{Request JSON gửi lên /momo/payment/request}
    \begin{tabular}{|p{4cm}|p{2.5cm}|p{7cm}|}
        \hline
        \textbf{Trường} & \textbf{Kiểu dữ liệu} & \textbf{Ý nghĩa} \\
        \hline
        partnerCode & String & Mã định danh đối tác do MoMo cấp. \\
        \hline
        orderId & String & Mã đơn hàng duy nhất trong hệ thống. \\
        \hline
        amount & Number & Số tiền thanh toán (VNĐ). \\
        \hline
        orderInfo & String & Thông tin mô tả đơn hàng. \\
        \hline
        redirectUrl & String & URL chuyển hướng sau khi thanh toán thành công. \\
        \hline
        ipnUrl & String & URL nhận thông báo trạng thái giao dịch từ MoMo (IPN). \\
        \hline
        requestType & String & Loại yêu cầu (thường là "captureWallet"). \\
        \hline
        signature & String & Chữ ký số xác thực dữ liệu. \\
        \hline
    \end{tabular}
\end{table}

\textit{Response trả về từ MoMo:}
\begin{table}[H]
    \centering

        \hline
        partnerCode & String & Mã đối tác MoMo cấp. \\
        \hline
        orderId & String & Mã đơn hàng gửi lên. \\
        \hline
        requestId & String & Mã request gửi lên MoMo. \\
        \hline

        resultCode & Number & 0: thành công, khác 0: lỗi. \\
        \hline
        message & String & Thông báo kết quả. \\
        \hline
        payType & String & Hình thức thanh toán. \\
        \hline
        responseTime & Number & Thời gian phản hồi (timestamp). \\

    \end{tabular}
\end{table}

\textbf{Lưu ý:} Các trường có thể thay đổi tuỳ theo tài liệu chính thức của MoMo từng thời điểm. Nên đối chiếu tài liệu mới nhất khi tích hợp.

\textbf{Lưu ý:} Các API đều có kiểm tra chữ ký số (signature) để đảm bảo an toàn giao dịch. Thông tin chi tiết về các trường dữ liệu, mã lỗi và quy trình xác thực tham khảo tài liệu chính thức của MoMo hoặc phần tài liệu kỹ thuật dự án.
