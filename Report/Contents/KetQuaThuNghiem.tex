\chapter{Kết quả thử nghiệm}

\section{Kết quả thử nghiệm \& đánh giá}

\subsection{Kiểm thử chức năng (Functional Testing)}

Phần này trình bày kết quả kiểm thử chức năng của hệ thống, bao gồm kiểm thử đơn vị và kiểm thử tích hợp.

\subsubsection{Kiểm thử đơn vị (Unit Tests)}

Kiểm thử đơn vị nhằm đảm bảo từng thành phần nhỏ (Controller, Service) hoạt động đúng khi được tách riêng khỏi phần còn lại của hệ thống. Hệ thống sử dụng \textbf{Jest} làm công cụ kiểm thử chính, với tổng cộng \textbf{23 tệp kiểm thử} bao phủ toàn bộ các module:

\begin{itemize}
    \item \texttt{auth}: Kiểm tra đăng ký, đăng nhập, làm mới token và xác thực JWT.
    \item \texttt{partners}: Kiểm tra đăng ký đối tác, cập nhật hồ sơ, chuyển đổi trạng thái xác minh.
    \item \texttt{admin}: Kiểm tra quy trình xét duyệt, xác định kết quả duyệt, ghi nhận lịch sử.
    \item \texttt{audit}: Kiểm tra ghi nhật ký kiểm toán tự động.
    \item \texttt{employees}, \texttt{products}, \texttt{categories}, \texttt{locations}, \texttt{service-tags}, \texttt{s3}: Kiểm tra các thao tác thêm/sửa/xóa và logic nghiệp vụ.
\end{itemize}

Để cô lập từng thành phần, các bài kiểm thử sử dụng đối tượng giả lập (mock) thay thế cho cơ sở dữ liệu thực. Mỗi bài kiểm thử tuân theo mô hình 3 bước: \textbf{Chuẩn bị} dữ liệu đầu vào, \textbf{Thực thi} hàm cần kiểm tra, và \textbf{Xác nhận} kết quả đầu ra.

\subsubsection{Kiểm thử tích hợp (End-to-End Tests)}

Kiểm thử tích hợp mô phỏng hành vi thực tế của người dùng bằng cách gửi các yêu cầu HTTP đến hệ thống đang chạy đầy đủ. Các bài kiểm thử được tổ chức trong thư mục \texttt{test/e2e/}, sử dụng thư viện \texttt{Supertest}:

\begin{itemize}
    \item \texttt{auth/}: Kiểm tra luồng hoàn chỉnh từ đăng ký $\rightarrow$ đăng nhập $\rightarrow$ làm mới token.
    \item \texttt{account/}: Kiểm tra cập nhật hồ sơ cá nhân.
    \item \texttt{categories/}, \texttt{employees/}, \texttt{products/}: Kiểm tra các API thêm/sửa/xóa từ đầu đến cuối.
\end{itemize}

Để hỗ trợ việc viết kiểm thử nhanh và nhất quán, hệ thống cung cấp các thư mục tiện ích:

\begin{itemize}
    \item \texttt{test/factories/}: Tạo dữ liệu mẫu tự động cho các bài kiểm thử.
    \item \texttt{test/fixtures/}: Dữ liệu cố định dùng chung.
    \item \texttt{test/mocks/}: Các đối tượng giả lập dùng chung giữa nhiều bài kiểm thử.
    \item \texttt{test/helpers/}: Các hàm tiện ích hỗ trợ khởi tạo và dọn dẹp môi trường kiểm thử.
\end{itemize}

\subsubsection{Công cụ và lệnh kiểm thử}

\begin{itemize}
    \item \textbf{Chạy kiểm thử đơn vị:} \texttt{npm test}
    \item \textbf{Chạy kiểm thử tích hợp:} \texttt{npm run test:e2e}
    \item \textbf{Xem độ bao phủ mã nguồn:} \texttt{npm run test:cov}
    \item \textbf{Kiểm tra chất lượng mã:} \texttt{npx eslint src/}
    \item \textbf{Định dạng mã nguồn:} \texttt{npx prettier --write src/}
\end{itemize}

\subsection{Đánh giá hiệu quả AI trong gợi ý dịch vụ}

Phần này trình bày phương pháp, tiêu chí và kết quả đánh giá cho ba thành phần cốt lõi của hệ thống AI:
\begin{itemize}
    \item Mô hình phân loại ý định (Intent Classifier).
    \item Dịch vụ trích xuất thực thể (NER Service).
    \item Hệ thống gợi ý (Recommender Service).
\end{itemize}

\subsubsection{Mô hình phân loại ý định (Intent Classifier)}

\textbf{Phương pháp và tiêu chí đánh giá:}

\begin{itemize}
    \item Quy trình: làm sạch dữ liệu, chia 80\% train -- 20\% test bằng stratified split.
    \item Mô hình: \texttt{SentenceTransformer(all-MiniLM-L6-v2)} để mã hóa văn bản, \texttt{LogisticRegression} để phân lớp nhị phân.
    \item Ngưỡng đạt yêu cầu:
    \begin{itemize}
        \item Accuracy, Macro F1: $\geq 0.85$.
        \item Precision và Recall cho lớp 1: $\geq 0.83$.
    \end{itemize}
\end{itemize}

\textbf{Kết quả định lượng:}

\begin{itemize}
    \item Tập test: 187 mẫu (94 mẫu lớp 0, 93 mẫu lớp 1).
    \item Ma trận nhầm lẫn được thể hiện ở Hình \ref{fig:intent_confusion_matrix}.
\end{itemize}

    \item Macro F1: trung bình F1 giữa các lớp, phản ánh độ cân bằng.

    \item Kết luận: đã đủ chất lượng để vận hành trong pipeline gợi ý dịch vụ.
\end{itemize}

\begin{itemize}

\subsubsection{Hệ thống gợi ý (Recommender Service)}

\begin{itemize}
    \item Hai kịch bản đánh giá độc lập:
    \begin{itemize}
        \item Chatbot: truy vấn hội thoại.
        \item Home: truy vấn dựa trên hồ sơ người dùng.
    \end{itemize}
    \item Bộ chỉ số: Hit@K, Recall@K, MRR@K, NDCG@K, HNR và thống kê latency.
    \item Kết quả Home dùng cấu hình \texttt{query\_mode=keyword}, \texttt{score\_mode=distance}.

\begin{itemize}
    \item Hit@3: tỷ lệ truy vấn có ít nhất 1 kết quả đúng trong top-3.
    \item Recall@3: mức bao phủ kết quả liên quan trong top-3.
    \item MRR@3: ưu tiên đúng kết quả liên quan ở vị trí càng cao càng tốt.
    \item NDCG@3: chất lượng xếp hạng có xét vị trí.
    \item HNR@3: tỷ lệ hard-negative lọt top-3 (càng thấp càng tốt).
\end{itemize}

\begin{itemize}
    \item Chatbot cho chất lượng truy xuất tốt hơn rõ rệt (đặc biệt Hit@3 và MRR@3).
    \item Home có lợi thế về tốc độ nhưng Recall thấp hơn đáng kể, còn bỏ sót kết quả liên quan.
    \item HNR@3 của cả hai kịch bản đều thấp ($<0.03$), cho thấy hệ thống lọc nhiễu tốt ở top đầu.
\end{itemize}
