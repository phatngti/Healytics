\chapter{Hiện thực}
\section{Các công nghệ liên quan}

\subsection{Mobile App Development (iOS/Android, Flutter)}
\begin{itemize}
    \item \textbf{Lý do chọn Flutter:} Là một framework đa nền tảng, cho phép phát triển ứng dụng cho cả iOS và Android từ một bộ mã nguồn duy nhất. Điều này giúp \textbf{tiết kiệm thời gian, chi phí} phát triển và bảo trì, đồng thời đảm bảo trải nghiệm người dùng nhất quán trên cả hai hệ điều hành. Hiệu năng của Flutter gần như tương đương với ứng dụng native.
    \item \textbf{Kiến trúc:} Áp dụng kiến trúc Clean Architecture hoặc Riverpod Architecture để tách biệt rõ ràng các tầng logic (Data, Domain, Presentation), giúp mã nguồn dễ kiểm thử và bảo trì.
    \item \textbf{Tính năng nổi bật}:
    \begin{itemize}
    \item \textbf{Impeller Rendering Engine}: Flutter biên dịch shader tùy chỉnh tại thời điểm build để đạt độ mượt 120Hz và không có JS bridge; React Native với New Architecture (Fabric + Skia) vẫn dùng cầu nối JS-to-native, bị giật khi animation nặng; Native (Metal/Vulkan) truy cập GPU trực tiếp nhưng cần tinh chỉnh riêng cho từng nền tảng.
    
    \item \textbf{Hot Reload Speed}: Flutter giữ trạng thái trong <100ms nhờ Dart AOT; React Native với Metro bundler reload trong ~1s nhưng mất trạng thái ở module native phức tạp; Native không có hot reload thật sự—rebuild toàn bộ mất 10s+ trong Xcode/Android Studio.
    
    \item \textbf{Widget/Component System}: Mô hình ``mọi thứ là widget'' của Flutter cho phép kết hợp vô hạn và UI chính xác đến từng pixel chỉ trong một ngôn ngữ (Dart); React Native dùng JSX cần thư viện bên thứ ba cho style nâng cao, dễ bất đồng bộ; Native (SwiftUI/Compose) kiểm soát hoàn toàn nhưng phải viết lại code cho iOS/Android.
    
    \item \textbf{Startup Time \& App Size}: Flutter khởi động lạnh <50ms nhờ AOT nhưng bundle engine ~50MB; React Native khởi động ~100ms với JS bundle, kích thước cơ bản nhỏ hơn (~30MB); Native khởi động nhanh nhất (<30ms) không có runtime nhúng.
    
    \item \textbf{Platform Channel/Bridge}: Kênh đồng bộ/bất đồng bộ của Flutter tích hợp API native (camera, BLE) với Dart an toàn kiểu; TurboModules của React Native giảm overhead cầu nối nhưng vẫn chỉ bất đồng bộ, khó debug; Native truy cập phần cứng trực tiếp qua SDK, độ trễ bằng 0.
    
    \item \textbf{Multi-Platform Code Reuse}: Flutter biên dịch thành binary native cho iOS, Android, web, desktop và nhúng với tỷ lệ tái sử dụng 90\%+; React Native mạnh trên mobile, cần React Native Web/Electron cho các nền khác (~70\% tái sử dụng); Native yêu cầu codebase Swift/Kotlin riêng, tái sử dụng <30\%.
    
    \item \textbf{Animation Performance}: Impeller + canvas 120Hz của Flutter cho 0 khung hình bị bỏ trong chuyển cảnh phức tạp; React Native ổn định ~60fps nhưng giật khi JS thread nặng; Native đảm bảo 120Hz với Core Animation/Compose, tốt nhất cho game.
    
    \item \textbf{OTA Updates}: Flutter dùng Shorebird/CodeMagic để vá lỗi tức thì không qua store; React Native dùng CodePush đẩy cập nhật JS trực tiếp; Native phải qua App Store/Google Play, chờ duyệt 1--7 ngày.
    
    \item \textbf{Ecosystem \& Third-Party Libs}: pub.dev của Flutter phát triển nhanh nhưng thiếu plugin niche, thường cần viết channel tùy chỉnh; React Native tận dụng npm với 2M+ package, hệ sinh thái phong phú nhất; Native dùng CocoaPods/Gradle chính thức, ổn định nhất nhưng bị khóa nền tảng.
\end{itemize}


\subsection{Web/Portal Development (Partner Portal, Admin Dashboard)}
\begin{itemize}
    \item \textbf{Lý do chọn Next.js:} Là một framework React hàng đầu, cung cấp hiệu năng vượt trội nhờ các cơ chế render linh hoạt (Server-Side Rendering - SSR, Static Site Generation - SSG). Điều này rất quan trọng cho trang Blog/CMS để tối ưu SEO. Việc phát triển Partner Portal và Admin Dashboard với nhiều biểu đồ, bảng biểu cũng được React hỗ trợ rất tốt.
    \item \textbf{Tính năng nổi bật}:
    \begin{itemize}
    \item \textbf{Server-Side Rendering (SSR) \& Static Site Generation (SSG)}: Next.js vượt trội với SSR/SSG/ISR kết hợp, tối ưu SEO và thời gian tải (<2s FCP); React cần cấu hình thủ công qua công cụ bên thứ ba như Remix, thiếu tối ưu tích hợp; Nuxt.js (Vue) tương đương với engine Nitro, khởi động lạnh ~2ms; Angular cần thư viện bổ sung cho SSR, hydration chậm; Gatsby tập trung SSG nhưng rebuild toàn site khi cập nhật, kém linh hoạt cho ứng dụng động.
    
    \item \textbf{React Server Components (RSC)}: Next.js tích hợp RSC nguyên bản trong App Router để xử lý logic chỉ server, giảm 50\%+ kích thước bundle và hỗ trợ streaming UI; React hỗ trợ RSC thử nghiệm nhưng thiếu khung kết nối; Remix dùng loader lồng nhau cho server fetching tương tự nhưng không phải RSC thật; Nuxt thiếu tương đương RSC, dựa vào Composition API của Vue; Angular mô phỏng qua zoneless detection nhưng gắn với MVC, kém chi tiết.
    
    \item \textbf{Turbopack Bundler}: Next.js dùng Turbopack (Rust) ở v15 cho HMR nhanh gấp 700 lần Webpack, ổn định dev/prod; React/Webpack chậm ~10s rebuild; Vite của Nuxt nhanh (~1s) nhưng kém tối ưu với app quy mô React lớn; Gatsby/Webpack chậm nhất cho build tăng dần; esbuild của Angular CLI nhanh nhưng thiếu edge caching như Turbopack.
    
    \item \textbf{App Directory \& Routing}: App Router của Next.js dùng file-based routing, hỗ trợ layout lồng nhau, route song song, interceptors mà không cần config; React Router cần thiết lập thủ công, dễ dư thừa code; Nuxt tự động routing tương tự nhưng chỉ dành cho Vue; Angular dùng module lazy-load có cấu trúc nhưng cứng nhắc, chậm với SPA; routing GraphQL của Gatsby mạnh cho static nhưng yếu với param động.
    
    \item \textbf{Server Actions}: Ổn định trong Next.js 15 cho mutation client-to-server trực tiếp không cần API, tăng bảo mật/hiệu suất; React không có sẵn, phải dùng custom hook; Remix cung cấp actions/loaders xử lý form nhưng dài dòng; Nuxt dùng composables tương tự nhưng kém liền mạch; Angular dùng services + NgRx để đồng bộ state, overhead nặng.
    
    \item \textbf{SEO \& Metadata Handling}: generateMetadata API của Next.js tự động xử lý OG tag/sitemap động, tăng 40\%+ khả năng crawl; React cần next/head hoặc React Helmet, thủ công từng trang; useHead của Nuxt tương đương, thân thiện Vue; Angular dùng Title/Meta services nhưng thiếu streaming; plugin Gatsby tối ưu SEO static nhưng khó với SEO động.
    
    \item \textbf{Performance \& Core Web Vitals}: Next.js đạt 90+ điểm Lighthouse nhờ ISR/edge caching, cải thiện 50--70\% FCP; React CSR thường 70--80 nếu không tinh chỉnh; Vapor Mode của Nuxt loại bỏ VDOM overhead, hiệu suất tương đương; Angular hiệu quả với doanh nghiệp nhưng hydration chậm; Gatsby xuất sắc static (giảm 95\% JS) nhưng thời gian rebuild ảnh hưởng CI/CD.
    
    \item \textbf{TypeScript Integration}: Next.js hỗ trợ TS zero-config với App Router suy luận kiểu, mượt mà cho codebase lớn; React TS tốt nhưng cần Create React App setup; Nuxt tự sinh type qua module; Angular là TS-first với kiểm tra nghiêm ngặt, nhưng đường cong học cao; hỗ trợ TS của Gatsby dựa plugin, không đồng nhất.
    
    \item \textbf{Ecosystem \& Deployment}: Next.js tận dụng 2M+ package npm + Vercel triển khai edge (zero-cold-start); hệ sinh thái React rộng nhưng phân mảnh; Nuxt có hệ sinh thái Vue gắn kết (Nuxt Hub/UI); Angular tập trung doanh nghiệp (Ivy compiler); Gatsby giàu plugin CMS nhưng phụ thuộc Vercel ít hơn Next.js.
    
    \item \textbf{Developer Experience (DX)}: Fast Refresh + Turbopack của Next.js giảm thời gian lặp lại xuống <1s, có AI SDK hỗ trợ prototype nhanh; HMR của React tốt nhưng thiếu routing/tối ưu tích hợp; module preset của Nuxt tăng tốc khởi tạo; Angular CLI scaffold nhanh nhưng cứng nhắc; Remix nhấn mạnh forms/actions cho DX thương mại điện tử.
\end{itemize}


\subsection{Server/Backend (NestJS \& PostgreSQL)}
\begin{itemize}
    \item \textbf{Lý do chọn NestJS:} Được xây dựng trên nền tảng NodeJS và TypeScript, NestJS cung cấp một kiến trúc module hóa, có cấu trúc chặt chẽ (lấy cảm hứng từ Angular). Điều này giúp việc xây dựng các API phức tạp trở nên \textbf{dễ quản lý, bảo trì và mở rộng}. TypeScript cũng giúp giảm thiểu lỗi và tăng cường tính an toàn của mã nguồn.
    \item \textbf{Database: PostgreSQL} là lựa chọn phù hợp nhất. PostgreSQL là hệ quản trị CSDL quan hệ mạnh mẽ, ổn định, có khả năng xử lý các truy vấn phức tạp và lượng dữ liệu lớn. Các tính năng như hỗ trợ JSONB rất hữu ích để lưu các dữ liệu có cấu trúc linh hoạt (ví dụ: cấu hình dịch vụ, câu hỏi cá nhân hóa).
    \item \textbf{Tính năng nổi bật:}
    \begin{itemize}
    \item \textbf{Modular Architecture}: NestJS áp dụng cấu trúc module tính năng với @Module() để tách biệt rõ ràng, lý tưởng cho ứng dụng doanh nghiệp; Express thiếu module tích hợp, dễ dẫn đến code rối; Spring Boot có module bằng annotation nhưng dài dòng với Java; Django dùng app-based module kiểu Python, đầy đủ tính năng nhưng kém linh hoạt với hệ sinh thái JS; Fastify nhẹ, không có module gốc, cần tổ chức thủ công.

    \item \textbf{Dependency Injection (DI)}: NestJS cung cấp DI phân cấp, an toàn kiểu qua @Injectable() và providers, giảm boilerplate và tăng khả năng test; Express không có DI sẵn, phải dùng thư viện như Inversify; Spring Boot với @Autowired là tiên phong nhưng nặng với cấu hình XML; Django truyền service thủ công, không có container DI thật; Fastify hỗ trợ plugin nhưng thiếu DI cấp framework.

    \item \textbf{TypeScript-First Design}: NestJS được xây dựng hoàn toàn cho TypeScript với suy luận kiểu và decorator, phát hiện lỗi tại compile-time; Express hỗ trợ TS qua add-on nhưng vẫn JS-centric; Spring Boot dùng kiểu tĩnh Java, cứng nhắc hơn TS; Django là Python động, type hints chỉ là bổ sung; Fastify có type TS nhưng tích hợp không sâu, dễ lỗi runtime.

    \item \textbf{Decorators \& Metadata}: NestJS dùng @Controller(), @Get(), @Body() để định nghĩa route và validation khai báo, lấy cảm hứng từ Angular; Express dùng hàm/middleware thuần, dài dòng; Spring Boot với @RestController/@RequestMapping tương tự nhưng cú pháp Java; Django dùng class-based hoặc function view, không có decorator; Fastify dùng schema route, ít metadata.

    \item \textbf{Built-in Testing Support}: NestJS tích hợp @nestjs/testing cho test e2e/unit với mock và override DI sẵn; Express cần setup Jest/Supertest thủ công; Spring Boot với @SpringBootTest toàn diện nhưng khởi động JVM chậm; Django có TestCase tích hợp nhưng chỉ cho Python; Fastify test dựa plugin, nhẹ nhưng thiếu cấu trúc như Nest.

    \item \textbf{Microservices \& Event-Driven}: NestJS hỗ trợ microservices nguyên bản qua @nestjs/microservices với TCP, Redis, Kafka, hybrid; Express cần setup thủ công; Spring Boot với Spring Cloud mạnh cho doanh nghiệp nhưng phức tạp; Django dùng Celery cho task, không phải microservices đầy đủ; Fastify tập trung HTTP, hạn chế về async messaging.

    \item \textbf{Performance \& Adapter Flexibility}: NestJS với Fastify adapter đạt 30k+ req/s (gấp 2 Express), bộ nhớ thấp (~15\% so Express, 1/5 Spring); Express nhanh cho app nhỏ nhưng nghẽn khi scale; Spring Boot tối ưu JVM cho tải nặng nhưng latency/bộ nhớ cao; Django bị giới hạn GIL về concurrency, chậm hơn Node; Fastify đạt 33k req/s nhưng Nest thêm ~8\% overhead để có cấu trúc.

    \item \textbf{Validation \& Pipes}: NestJS dùng class-validator với pipes @nestjs/class-transformer để tự động validate DTO khai báo; Express dùng Joi/express-validator thủ công; Spring Boot với @Valid và Bean Validation tương tự nhưng dùng Java bean; Django xử lý validation qua forms/serializers; Fastify schema hiệu suất cao nhưng tích hợp kém.

    \item \textbf{Guards, Interceptors \& Pipes}: NestJS hỗ trợ tính năng AOP qua @UseGuards, @UseInterceptors cho auth/logging/biến đổi; Express dùng middleware chain tương tự nhưng toàn cục; Spring Boot có AOP/Interceptors mạnh cho doanh nghiệp; Django middleware tuyến tính, kém chi tiết; Fastify dùng hook nhanh nhưng chỉ hook, không có guard.

    \item \textbf{Ecosystem \& Community}: NestJS tận dụng 2M+ package npm với 60k+ sao GitHub, tài liệu/plugin phát triển mạnh; Express có hệ sinh thái rộng nhưng không cấu trúc; Spring Boot trưởng thành cho doanh nghiệp (Maven/Gradle); Django mạnh về data/ML với PyPI; Fastify cộng đồng nhỏ hơn nhưng plugin hiệu suất cao.
\end{itemize}



\subsection{Hiện thực hệ thống RAG Chatbot}

Phần hiện thực tập trung mô tả chi tiết quá trình xây dựng và triển khai hệ thống Chatbot tích hợp Retrieval-Augmented Generation (RAG). Khác với phần thiết kế đã trình bày ở mục trước, nội dung dưới đây mô tả trực tiếp cách các thành phần được cài đặt, cách chúng vận hành trong môi trường thực tế, cùng các cơ chế xử lý dữ liệu, truy xuất và sinh câu trả lời dựa trên mã nguồn Python của hệ thống.

\subsubsection{Tổ chức mã nguồn}

Hệ thống được triển khai dưới dạng một ứng dụng Python theo cấu trúc module hoá. Các thành phần được tách riêng thành nhiều file nhằm đảm bảo tính bảo trì, dễ kiểm thử và dễ mở rộng. Cấu trúc thư mục chính bao gồm:

\begin{itemize}
    \item \texttt{loaders/}: Các bộ tải tài liệu (PDF, text).
    \item \texttt{processors/}: Bộ chia nhỏ văn bản và bộ làm sạch dữ liệu.
    \item \texttt{embeddings/}: Mã thực thi mô hình embedding.
    \item \texttt{vectorstore/}: Triển khai FAISS cho lưu trữ và truy vấn vector.
    \item \texttt{retrieval/}: Bộ truy xuất top-k đoạn liên quan.
    \item \texttt{prompt/}: Xây dựng format prompt hoàn chỉnh.
    \item \texttt{chatbot/}: Pipeline chính kết nối truy xuất và mô hình sinh.
\end{itemize}

Việc tách thành module giúp các chức năng hoạt động độc lập và giảm rủi ro trùng lặp logic.

\subsubsection{Hiện thực xử lý tài liệu}

\paragraph{Tải và làm sạch tài liệu}

Thành phần tải tài liệu được lập trình để quét nhiều định dạng.  
Đối với PDF, hệ thống sử dụng bộ giải mã dựa trên PyMuPDF nhằm trích xuất nội dung văn bản theo từng trang. Sau khi tải xong, dữ liệu thô được chuyển qua lớp xử lý để loại bỏ:

\begin{itemize}
    \item ký tự thừa, ký tự xuống dòng không cần thiết;
    \item khoảng trắng dư thừa;
    \item các ký tự không hỗ trợ mã hoá UTF-8.
\end{itemize}

Kết quả trả về là một chuỗi văn bản sạch có thể đưa vào bước chia đoạn.

\paragraph{Chia nhỏ văn bản}

Khối chia nhỏ văn bản được hiện thực bằng một lớp chuyên dụng. Mã nguồn thực thi thuật toán chia theo độ dài token cố định (thường 300–400 token). Các đoạn sinh ra overlapped một phần để tránh mất ngữ nghĩa.

Trong mã, bộ đếm token được thực hiện bằng tokenizer của mô hình embedding, đảm bảo số token đo được hoàn toàn trùng khớp với môi trường truy xuất và embedding.  
Kết quả được lưu dưới dạng danh sách JSON gồm nội dung đoạn và metadata đi kèm như tên file, trang số và chỉ số đoạn.

\subsubsection{Hiện thực mô hình embedding}

Hệ thống sử dụng mô hình Sentence Transformers được tải từ HuggingFace. Lớp embedding thực thi các bước:

\begin{enumerate}
    \item tải mô hình vào CPU hoặc GPU tùy khả năng phần cứng;
    \item mã hóa văn bản thành vector có kích thước cố định;
    \item chuẩn hóa vector về đơn vị (L2 normalize) trước khi lưu vào FAISS.
\end{enumerate}

Cơ chế xử lý hàng loạt (batch encoding) được tích hợp nhằm tăng hiệu năng khi embedding số lượng lớn đoạn.

\subsubsection{Cơ sở dữ liệu vector FAISS}

Vector sau khi được tạo sẽ được lưu vào chỉ mục FAISS dạng \texttt{IndexFlatIP}, tức sử dụng inner-product để tìm kiếm. Do các vector đã được chuẩn hóa, inner-product tương đương với cosine similarity.

Quá trình xây dựng chỉ mục bao gồm:

\begin{itemize}
    \item khởi tạo chỉ mục rỗng;
    \item chèn toàn bộ vector vào chỉ mục;
    \item lưu thêm metadata tương ứng trong một bảng ánh xạ riêng.
\end{itemize}

Chỉ mục được ghi xuống đĩa để tái sử dụng ở các lần chạy sau, giảm thời gian khởi động hệ thống.

\subsubsection{Hiện thực bộ truy xuất}

Tại thời điểm nhận câu hỏi từ người dùng, pipeline thực hiện:

\begin{enumerate}
    \item embedding câu hỏi thành vector cùng kích thước với vector tài liệu;
    \item gửi vector vào chỉ mục FAISS để truy vấn top-k (thường k = 5);
    \item truy xuất lại nội dung các đoạn thông qua metadata;
    \item xây dựng chuỗi ngữ cảnh bằng cách ghép các đoạn theo thứ tự giảm dần mức độ liên quan.
\end{enumerate}

Để tăng mức chính xác, mã nguồn có thêm bước lọc theo ngưỡng tương đồng. Nếu độ tương đồng thấp hơn mức tối thiểu, hệ thống trả về nhắc nhở rằng không tìm thấy thông tin phù hợp.

\subsubsection{Xây dựng prompt đầu vào}

Thành phần xây dựng prompt hiện thực đầy đủ 4 phần:

\begin{itemize}
    \item vai trò hệ thống (system role);
    \item ngữ cảnh lấy từ RAG;
    \item câu hỏi gốc;
    \item hướng dẫn ràng buộc.
\end{itemize}

Mã nguồn chủ động tối ưu để tránh vượt quá giới hạn token của mô hình bằng cách:

\begin{itemize}
    \item cắt bỏ các đoạn quá dài,
    \item ưu tiên giữ các đoạn có độ tương đồng cao nhất,
    \item chuẩn hóa định dạng ngữ cảnh dưới dạng bullet hoặc paragraph ngắn.
\end{itemize}

\subsubsection{Tích hợp mô hình ngôn ngữ}

Mô hình ngôn ngữ lớn được load trong chế độ suy luận lượng tử 4-bit, tận dụng thư viện HuggingFace Transformers kết hợp với accelerate và bitsandbytes. Thành phần hiện thực cung cấp:

\begin{itemize}
    \item cơ chế sinh văn bản theo chế độ greedy hoặc sampling;
    \item kiểm soát tối đa token đầu ra nhằm tránh sinh tràn;
    \item dừng sinh khi gặp các token kết thúc hoặc mẫu định dạng không hợp lệ.
\end{itemize}

Toàn bộ quá trình sinh được đóng gói thành một hàm duy nhất để dễ dàng gọi trong pipeline chatbot.

\subsubsection{Xử lý phản hồi và hậu xử lý}

Sau khi mô hình trả lời, hệ thống áp dụng chuỗi hậu xử lý:

\begin{itemize}
    \item loại bỏ token đặc biệt của mô hình;
    \item chuẩn hóa xuống dòng, dấu câu;
    \item cắt bỏ các phần nằm ngoài định dạng yêu cầu;
    \item kiểm tra câu trả lời có bám sát ngữ cảnh hay không.
\end{itemize}

Nếu mô hình sinh câu trả lời không liên quan đến ngữ cảnh truy xuất, hệ thống tự động trả về thông báo trung tính để tránh rủi ro ảo giác.

\subsubsection{Pipeline tổng hợp của Chatbot}

Pipeline cuối cùng được hiện thực trong module \texttt{chatbot/service.py}. Tại đây toàn bộ các bước được kết nối theo trình tự:

\begin{enumerate}
    \item nhận câu hỏi từ người dùng;
    \item embedding câu hỏi;
    \item truy xuất đoạn liên quan từ FAISS;
    \item xây dựng prompt hoàn chỉnh;
    \item gửi prompt vào mô hình sinh;
    \item trả về câu trả lời sau khi hậu xử lý.
\end{enumerate}

Mỗi bước đều có logging chi tiết giúp theo dõi hoạt động chatbot và hỗ trợ gỡ lỗi trong môi trường triển khai thực tế.




\subsection{Triển khai hệ thống gợi ý}

\subsubsection{Cấu trúc dữ liệu}

Hệ thống được triển khai với cấu trúc dữ liệu rõ ràng, bao gồm:

\begin{itemize}
    \item \textbf{Services Database}: File JSON chứa 10 dịch vụ mẫu về sức khỏe
    \item \textbf{Vector Database}: ChromaDB persistent storage tại \texttt{data/processed/vectordb/}
    \item \textbf{Embedding Model}: \texttt{sentence-transformers/paraphrase-multilingual-mpnet-base-v2}
\end{itemize}

Mỗi service trong database chứa các trường:
\begin{itemize}
    \item \texttt{id}: Unique identifier (VD: SV001, SV002)
    \item \texttt{name}: Tên dịch vụ
    \item \texttt{description}: Mô tả chi tiết (100-200 từ)
    \item \texttt{category}: Phân loại (y\_te, dinh\_duong, the\_thao, tam\_ly, phuc\_hoi)
    \item \texttt{tags}: Danh sách từ khóa (4-6 tags)
    \item \texttt{type}: Loại dịch vụ (consultation, exercise, nutrition, etc.)
\end{itemize}

\subsubsection{Offline Processing}

\paragraph{Document Construction} \\
Mỗi service được chuyển đổi thành một document text có cấu trúc:

\begin{quote}
\texttt{Tên dịch vụ: [name]}\\
\texttt{Mô tả: [description]}\\
\texttt{Danh mục: [category]}\\
\texttt{Từ khóa: [tag1, tag2, tag3, ...]}
\end{quote}

\paragraph{Vector Database Building} \\
Quá trình build vector database bao gồm:
\begin{enumerate}
    \item Load 10 services từ \texttt{data/raw/services.json}
    \item Tạo 10 documents tương ứng
    \item Embedding 10 documents thành 10 vectors (768 chiều mỗi vector)
    \item Index vào ChromaDB với cosine similarity metric
    \item Lưu persistent tại \texttt{data/processed/vectordb/}
\end{enumerate}

Kích thước vector database: $10 \times 768 = 7,680$ dimensions tổng cộng.

\subsubsection{Online Processing}

\paragraph{Home Recommender} \\

\textbf{Input Format:}
\begin{itemize}
    \item \texttt{user\_profile} (dict, optional):
    \begin{itemize}
        \item \texttt{health\_conditions}: List[str] - VD: ["tim mạch", "đái tháo đường"]
        \item \texttt{interests}: List[str] - VD: ["yoga", "dinh dưỡng"]
        \item \texttt{goals}: List[str] - VD: ["giảm cân", "kiểm soát đường huyết"]
    \end{itemize}
    \item \texttt{selected\_services} (List[str], optional): ["SV001", "SV003"]
    \item \texttt{top\_k} (int): Số lượng gợi ý mong muốn (default: 5)
\end{itemize}

\textbf{Processing Pipeline:}
\begin{enumerate}
    \item \textbf{Query Building}: Nối tất cả elements từ health\_conditions, interests, goals, và tags từ selected\_services thành một string
    \item \textbf{Embedding}: Query string → 768-dim vector
    \item \textbf{Vector Search}: ChromaDB query với cosine similarity
    \item \textbf{Filtering}: Loại bỏ services trong selected\_services, filter theo threshold (0.2)
    \item \textbf{Ranking}: Sắp xếp theo similarity score giảm dần
    \item \textbf{Explanation}: Generate lý do dựa trên matching tags và profile
\end{enumerate}

\textbf{Output Format:}
\begin{itemize}
    \item \texttt{recommendations}: List[Dict] - Danh sách services được gợi ý
    \begin{itemize}
        \item \texttt{service\_id}: str - VD: "SV002"
        \item \texttt{name}: str - Tên dịch vụ
        \item \texttt{description}: str - Mô tả đầy đủ
        \item \texttt{category}: str - Danh mục
        \item \texttt{tags}: List[str] - Tags
        \item \texttt{type}: str - Loại dịch vụ
        \item \texttt{similarity\_score}: float - Score trong [0, 1]
        \item \texttt{reason}: str - Lý do gợi ý
    \end{itemize}
    \item \texttt{total}: int - Tổng số gợi ý
\end{itemize}

\paragraph{Chatbot Recommender} \\

\textbf{Input Format:}
\begin{itemize}
    \item \texttt{query} (str): Câu hỏi tự nhiên từ người dùng
    \begin{itemize}
        \item VD: "Tôi bị stress và lo âu, cần tìm dịch vụ giúp giảm căng thẳng"
        \item VD: "Có khóa học yoga nào phù hợp với người đau lưng không?"
    \end{itemize}
    \item \texttt{top\_k} (int): Số lượng gợi ý (default: 5)
\end{itemize}

\textbf{Processing Pipeline:}
\begin{enumerate}
    \item \textbf{Query Embedding}: Câu query tự nhiên → 768-dim vector (không cần preprocessing)
    \item \textbf{Vector Search}: Tìm kiếm trong ChromaDB
    \item \textbf{Filtering}: Áp dụng similarity threshold (0.2)
    \item \textbf{Ranking}: Sắp xếp theo similarity score
    \item \textbf{Explanation}: Match tags trong query với tags của service
\end{enumerate}

\textbf{Output Format:}
\begin{itemize}
    \item \texttt{query}: str - Query gốc của người dùng
    \item \texttt{recommendations}: List[Dict] - Giống format Home Recommender
    \item \texttt{total}: int - Tổng số gợi ý
\end{itemize}

\subsubsection{Đánh giá hiệu năng}

\paragraph{Độ phức tạp tính toán} \\

\begin{itemize}
    \item \textbf{Embedding}: $O(L)$ với $L$ là độ dài query (thời gian constant cho mô hình BERT)
    \item \textbf{Vector Search}: $O(\log N)$ với HNSW index, $N$ = 10 services
    \item \textbf{Total latency}: < 100ms trên GPU, < 500ms trên CPU
\end{itemize}

\paragraph{Memory Requirements} \\

\begin{itemize}
    \item \textbf{Embedding Model}: ~420MB (model weights)
    \item \textbf{Vector Database}: ~80KB (10 services × 768 dims × 4 bytes)
    \item \textbf{Runtime Memory}: ~600MB total
\end{itemize}

\paragraph{Scalability} \\

Hệ thống có thể scale hiệu quả:
\begin{itemize}
    \item \textbf{1,000 services}: Latency < 100ms, Memory ~8MB cho vectors
    \item \textbf{10,000 services}: Latency < 150ms, Memory ~80MB cho vectors
    \item \textbf{100,000 services}: Latency < 200ms, Memory ~800MB cho vectors
\end{itemize}

HNSW index đảm bảo độ phức tạp $O(\log N)$ cho search, phù hợp với database lớn.

\subsubsection{Khả năng mở rộng}

Hệ thống được thiết kế để dễ dàng mở rộng theo các hướng:

\begin{itemize}
    \item \textbf{Incremental Updates}: Thêm services mới vào vector database mà không cần rebuild toàn bộ
    \item \textbf{Multi-language Support}: Mô hình embedding hiện tại đã hỗ trợ 50+ ngôn ngữ
    \item \textbf{Personalization}: Có thể thêm user embedding để cá nhân hóa sâu hơn
    \item \textbf{Hybrid Ranking}: Kết hợp semantic search với collaborative filtering hoặc popularity scores
    \item \textbf{Real-time Feedback}: Tích hợp user feedback để cải thiện ranking theo thời gian
\end{itemize}

\subsection*{Tầng Giao Diện (Presentation Layer)}
Đây là tầng tương tác trực tiếp với người dùng cuối (tương ứng với Client Layer trong bản thiết kế trừu tượng).

\begin{itemize}
    \item \textbf{Công nghệ:} Flutter.
    \item \textbf{Mô tả:} Sử dụng Flutter làm công nghệ chủ đạo (Cross-platform) để xây dựng cả 3 ứng dụng client:
    \begin{itemize}
        \item \textit{User App:} Ứng dụng di động dành cho người dùng phổ thông.
        \item \textit{Admin Panel:} Trang quản trị hệ thống.
        \item \textit{Provider Portal:} Cổng thông tin dành cho các nhà cung cấp dịch vụ y tế/làm đẹp.
    \end{itemize}
    \item \textbf{Giao thức:} Các ứng dụng này giao tiếp với hệ thống backend thông qua RESTful API và WebSocket (dành cho các tính năng thời gian thực).
\end{itemize}

\subsection*{Tầng Kết Nối \& API (Connection \& API Layer)}
Đóng vai trò là cổng vào duy nhất (Entry point) cho mọi yêu cầu từ phía Client.

\begin{itemize}
    \item \textbf{Công nghệ:} Kong API Gateway.
    \item \textbf{Mô tả:} Thay thế cho khối "API Gateway" trừu tượng. Kong chịu trách nhiệm:
    \begin{itemize}
        \item Routing (định tuyến) request đến đúng microservice.
        \item Load balancing (cân bằng tải).
        \item Authentication/Authorization cơ bản (xác thực sơ bộ).
        \item Rate limiting (giới hạn băng thông) để bảo vệ hệ thống.
    \end{itemize}
\end{itemize}

\subsection*{Tầng Server \& Nghiệp Vụ (Server/Business Layer)}
Đây là lõi của hệ thống, được triển khai theo kiến trúc Microservices.

\begin{itemize}
    \item \textbf{Hạ tầng:} Compute Engine (Google Cloud Platform - GCP). Các dịch vụ được triển khai trên các máy ảo (VM) hoặc Container chạy trên VM.
    \item \textbf{Các dịch vụ thành phần:}
    \begin{itemize}
        \item \textit{Authenticate Service:} Xử lý đăng ký, đăng nhập, cấp phát token (JWT).
        \item \textit{Core Healthcare Service:} Xử lý nghiệp vụ chính như đặt lịch hẹn (Appointments), quản lý hồ sơ y tế, thanh toán.
        \item \textit{AI Service:} Chịu trách nhiệm về các tính năng thông minh (Chatbot, Recommendation System). Dịch vụ này giao tiếp chặt chẽ với Vector Database.
        \item \textit{Notification Service:} Quản lý việc gửi thông báo đẩy (Push) và Email.
    \end{itemize}
\end{itemize}

\subsection*{Tầng Dữ Liệu \& Hạ Tầng (Data \& Infrastructure Layer)}
Tầng này lưu trữ và quản lý luồng dữ liệu, cụ thể hóa các khối database trừu tượng:

\begin{itemize}
    \item \textbf{Relational Database (SQL):} Sử dụng \textbf{PostgreSQL}. Dùng để lưu trữ dữ liệu có cấu trúc (thông tin user, đơn hàng, lịch hẹn).
    \item \textbf{Vector Database:} Sử dụng \textbf{FAISS} (Facebook AI Similarity Search). Dùng để lưu trữ vector embeddings phục vụ cho AI Service (tìm kiếm ngữ nghĩa, gợi ý).
    \item \textbf{Cache:} Sử dụng \textbf{Redis}. Giúp tăng tốc độ truy xuất dữ liệu thường xuyên sử dụng và giảm tải cho Database chính.
    \item \textbf{Message Queue:} Sử dụng \textbf{RabbitMQ}. Đảm bảo giao tiếp bất đồng bộ (asynchronous) giữa các microservices (ví dụ: Core Service gửi sự kiện đến Notification Service mà không cần chờ đợi).
    \item \textbf{Object Storage:} Sử dụng \textbf{MinIO}. Giải pháp lưu trữ đối tượng (tương thích S3) dùng để lưu trữ file tĩnh, hình ảnh, tài liệu y tế (thay thế cho khối Cloud Storage trừu tượng).
\end{itemize}

\subsection*{Tầng Tích Hợp Bên Thứ Ba (Third-Party Integrations Layer)}
Kết nối với các dịch vụ bên ngoài để mở rộng tính năng.

\begin{itemize}
    \item \textbf{Cổng thanh toán (Payment Methods):} Tích hợp đa dạng bao gồm MoMo, Stripe (quốc tế), và VNPay (nội địa Việt Nam).
    \item \textbf{Dịch vụ thông báo (App Notification Service):}
    \begin{itemize}
        \item \textit{APNs (Apple Push Notification service):} Dành riêng cho thiết bị iOS.
        \item \textit{Firebase (FCM):} Dành cho Android và quản lý thông báo chung.
    \end{itemize}
\end{itemize}
